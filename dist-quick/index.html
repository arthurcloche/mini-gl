🎨 Basic Shader Template

#version 300 es
precision highp float;

// Built-in uniforms
uniform vec2 glResolution;
uniform float glTime;
uniform vec3 glMouse;

// Built-in varyings
in vec2 glCoord;
in vec2 glUV;

// Output
out vec4 fragColor;

void main() {
    // Normalized coordinates (0-1)
    vec2 uv = glUV;
    
    // Aspect-corrected centered coordinates (-1 to 1)
    vec2 coord = (glCoord - glResolution * 0.5) / min(glResolution.x, glResolution.y);
    
    // Your shader code here
    vec3 color = vec3(uv.x, uv.y, sin(glTime) * 0.5 + 0.5);
    
    fragColor = vec4(color, 1.0);
}

🔧 Built-in Uniforms & Varyings

// Always available - no declaration needed
uniform vec2 glResolution;  // Canvas size in pixels (e.g., 1024.0, 1024.0)
uniform float glTime;       // Time in seconds * 0.01 (slow animation)
uniform vec3 glMouse;       // xy: mouse position (0-1), z: click state (0/1)
uniform float glFrame;      // Frame counter

in vec2 glCoord;           // Fragment position in pixels
in vec2 glUV;              // Normalized position (0.0 - 1.0)

🖼️ Texture Input Templates

Single Texture Input:

uniform sampler2D glTexture;

void main() {
    vec4 tex = texture(glTexture, glUV);
    fragColor = tex;
}

Multiple Texture Inputs:

uniform sampler2D uTextureA;
uniform sampler2D uTextureB;

void main() {
    vec4 texA = texture(uTextureA, glUV);
    vec4 texB = texture(uTextureB, glUV);
    fragColor = mix(texA, texB, 0.5);
}

✨ Common Effects

Distortion:

vec2 distortedUV = glUV;
distortedUV.x += sin(glUV.y * 10.0 + glTime) * 0.1;
distortedUV.y += cos(glUV.x * 10.0 + glTime) * 0.1;
vec4 tex = texture(glTexture, distortedUV);

Chromatic Aberration:

float aberration = 0.005;
float r = texture(glTexture, glUV + vec2(aberration, 0.0)).r;
float g = texture(glTexture, glUV).g;
float b = texture(glTexture, glUV - vec2(aberration, 0.0)).b;
fragColor = vec4(r, g, b, 1.0);

Vignette:

vec2 center = glUV - 0.5;
float vignette = 1.0 - dot(center, center) * 2.0;
vignette = clamp(vignette, 0.0, 1.0);
fragColor.rgb *= vignette;

Pixelation:

float pixels = 64.0;
vec2 pixelUV = floor(glUV * pixels) / pixels;
vec4 tex = texture(glTexture, pixelUV);

🎛️ Custom Uniform Examples

// Declare custom uniforms (add these to Properties panel)
uniform float uSpeed;      // 0.0 - 1.0 from slider
uniform float uIntensity;  // 0.0 - 1.0 from slider  
uniform float uScale;      // 0.0 - 1.0 from slider
uniform vec3 uColor;       // RGB color picker (future)

// Example usage
void main() {
    float time = glTime * uSpeed;
    vec2 scaledUV = (glUV - 0.5) * uScale + 0.5;
    
    float wave = sin(scaledUV.x * 10.0 + time) * uIntensity;
    vec3 color = vec3(wave);
    
    fragColor = vec4(color, 1.0);
}

🔨 Useful Helper Functions

// Rotation matrix
mat2 rotate2D(float angle) {
    float s = sin(angle);
    float c = cos(angle);
    return mat2(c, -s, s, c);
}

// Simple noise
float random(vec2 st) {
    return fract(sin(dot(st.xy, vec2(12.9898, 78.233))) * 43758.5453123);
}

// Smooth min (blend shapes)
float smin(float a, float b, float k) {
    float h = clamp(0.5 + 0.5 * (b - a) / k, 0.0, 1.0);
    return mix(b, a, h) - k * h * (1.0 - h);
}

// Distance field circle
float circle(vec2 p, float r) {
    return length(p) - r;
}

// Distance field box
float box(vec2 p, vec2 size) {
    vec2 d = abs(p) - size;
    return length(max(d, 0.0)) + min(max(d.x, d.y), 0.0);
}

🎨 Color Utilities

// HSV to RGB conversion
vec3 hsv2rgb(vec3 c) {
    vec4 K = vec4(1.0, 2.0 / 3.0, 1.0 / 3.0, 3.0);
    vec3 p = abs(fract(c.xxx + K.xyz) * 6.0 - K.www);
    return c.z * mix(K.xxx, clamp(p - K.xxx, 0.0, 1.0), c.y);
}

// RGB to grayscale
float luminance(vec3 color) {
    return dot(color, vec3(0.299, 0.587, 0.114));
}

// Contrast adjustment
vec3 contrast(vec3 color, float amount) {
    return (color - 0.5) * amount + 0.5;
}

💡 Quick Tips

  • Use glUV for texture coordinates (0-1 range)
  • Use glCoord for pixel-perfect effects
  • Multiply glTime to control animation speed
  • Always output to fragColor with alpha = 1.0
  • Use clamp() to keep values in 0-1 range
  • Test shaders with the "test" button before running

miniGL Editor - Quick Guide

📖 Overview

miniGL Editor is a node-based shader editor. Create visual effects by connecting nodes that represent textures, shaders, and effects. The output is rendered in real-time to the preview canvas.

🔄 Basic Workflow

  1. Add nodes from the Node Palette (drag or click Add)
  2. Connect nodes in the Node Graph (drag from output to input)
  3. Edit shader code by double-clicking shader nodes
  4. Adjust properties in the Node Properties panel
  5. Set the output node using the dropdown above the graph

🎨 Panels Guide

Scene Graph - Shows all nodes in your project. Click to select, use action buttons to rename/copy/delete.
Node Palette - Available node types organized by category. Drag to graph or click Add.
Preview - Real-time render output. Use controls to pause, reset, snapshot, or record.
Node Graph - Visual node connections. Drag nodes to move, drag ports to connect.
Properties - Settings for selected node. Sliders for uniforms, inputs for values.

🔲 Node Types

Texture - Load images from URLs or files
Text - Render text to texture
Shader - Custom GLSL fragment shader
Blend - Mix two inputs (screen/multiply/overlay)
Feedback - Create trailing/echo effects
Effects - Grayscale, Blur, Lens Distortion

✏️ Editing Shaders

Double-click any shader node to open the code editor. Shaders use GLSL ES 3.0.

Built-in uniforms:

  • vec2 glResolution - Canvas size in pixels
  • float glTime - Time in seconds (multiplied by 0.01)
  • vec3 glMouse - Mouse position (xy) and click state (z)
  • vec2 glCoord - Fragment coordinate
  • vec2 glUV - Normalized UV coordinates (0-1)

🎛️ Adding Custom Uniforms & Textures

For custom uniforms:

  1. Declare in your shader: uniform float myValue;
  2. Add to uniforms in Properties panel with matching name
  3. The slider will control the value (range 0-1 by default)

For texture inputs:

  1. Declare sampler: uniform sampler2D myTexture;
  2. Connect a Texture node to your shader node
  3. Sample in shader: texture(myTexture, glUV)

⚠️ Important: Uniform and texture names in your shader code must match exactly with the names you define in the Properties panel.

💡 Tips

  • Use Force Render if preview stops updating
  • Click test in shader editor to check for compilation errors
  • The console shows shader errors and logs
  • Nodes auto-connect when dragged near compatible ports
  • Use snapshot to save PNG images of your creation
  • Record videos up to 30 seconds with the record button
